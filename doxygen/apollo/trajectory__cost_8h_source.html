50  int CompareTo(const ComparableCost &other) const {
51  constexpr double kEpsilon = 1e-12;
52  if ((has_collision || out_of_boundary) &&
53  !(other.has_collision || other.out_of_boundary)) {
54  return 1;
55  } else if (!(has_collision || out_of_boundary) &&
56  (other.has_collision || other.out_of_boundary)) {
57  return -1;
58  } else if (std::fabs(safety_cost + smoothness_cost - other.safety_cost -
59  other.smoothness_cost) < kEpsilon) {
60  return 0;
61  } else if (safety_cost + smoothness_cost >
62  other.safety_cost + other.smoothness_cost) {
63  return 1;
64  } else {
65  return -1;
66  }
67  }
68  ComparableCost &operator+(const ComparableCost &other) {
69  has_collision = has_collision || other.has_collision;
70  out_of_boundary = out_of_boundary || other.out_of_boundary;
71  safety_cost += other.safety_cost;
72  smoothness_cost += other.smoothness_cost;
73  return *this;
74  }
75  ComparableCost &operator+=(const ComparableCost &other) {
76  has_collision = has_collision || other.has_collision;
77  out_of_boundary = out_of_boundary || other.out_of_boundary;
78  safety_cost += other.safety_cost;
79  smoothness_cost += other.smoothness_cost;
80  return *this;
81  }
82  bool operator>(const ComparableCost &other) const {
83  return this->CompareTo(other) > 0;
84  }
85  bool operator>=(const ComparableCost &other) const {
86  return this->CompareTo(other) >= 0;
87  }
88  bool operator<(const ComparableCost &other) const {
89  return this->CompareTo(other) < 0;
90  }
91  bool operator<=(const ComparableCost &other) const {
92  return this->CompareTo(other) <= 0;
93  }
