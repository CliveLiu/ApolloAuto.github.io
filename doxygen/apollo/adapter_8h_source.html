397  // HasSequenceNumber returns true if the data type is proto and has
398  // header.sequence_num.
399  template <typename InputMessageType>
400  static bool HasSequenceNumber(
401  enable_if_t<
402  std::is_base_of<google::protobuf::Message, InputMessageType>::value,
403  InputMessageType>* message = nullptr) {
404  using gpf = google::protobuf::FieldDescriptor;
405  InputMessageType sample;
406  auto descriptor = sample.GetDescriptor();
407  auto header_descriptor = descriptor->FindFieldByName("header");
408  if (header_descriptor == nullptr ||
409  header_descriptor->cpp_type() != gpf::CPPTYPE_MESSAGE) {
410  return false;
411  }

413  auto sequence_num_descriptor =
414  header_descriptor->message_type()->FindFieldByName("sequence_num");
415  return sequence_num_descriptor != nullptr &&
416  sequence_num_descriptor->cpp_type() == gpf::CPPTYPE_UINT32;
417  }

427  // DumpMessage dumps the message to a file to
428  // /tmp/<adapter_name>/<name>.pb.txt, where the message is in ASCII
429  // mode and <name> is the .header().sequence_num() of the message.
430  template <typename InputMessageType>
431  bool DumpMessage(const enable_if_t<std::is_base_of<google::protobuf::Message,
432  InputMessageType>::value,
433  InputMessageType>& message) {
434  using google::protobuf::Message;
435  auto descriptor = message.GetDescriptor();
436  auto header_descriptor = descriptor->FindFieldByName("header");
437  if (header_descriptor == nullptr) {
438  ADEBUG << "Fail to find header field in pb.";
439  return false;
440  }
441  const Message& header = message.GetReflection()->GetMessage(
442  *static_cast<const Message*>(&message), header_descriptor);
443  auto seq_num_descriptor =
444  header.GetDescriptor()->FindFieldByName("sequence_num");
445  if (seq_num_descriptor == nullptr) {
446  ADEBUG << "Fail to find sequence_num field in pb.";
447  return false;
448  }
449  uint32_t sequence_num =
450  header.GetReflection()->GetUInt32(header, seq_num_descriptor);
451  return util::SetProtoToASCIIFile(
452  message, util::StrCat(dump_path_, "/", sequence_num, ".pb.txt"));
453  }

475  // Don't try to copy data and enqueue if the message_num is 0
